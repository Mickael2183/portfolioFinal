Convertisseur Temporel

                                        # Convertisseur de durées temporelles
def convert_seconds(seconds):
    minutes = seconds // 60
    remaining_seconds = seconds % 60
    hours = minutes // 60
    remaining_minutes = minutes % 60
    days = hours // 24
    remaining_hours = hours % 24
    months = days // 30
    remaining_days = days % 30
    years = months // 12
    remaining_months = months % 12
    centuries = years // 100
    remaining_years = years % 100
    millennia = centuries // 10
    remaining_centuries = centuries % 10
    
    print(f"{millennia} millénaire(s) {remaining_centuries} siècle(s) {remaining_years} année(s), {remaining_months} mois, {remaining_days} jour(s), {remaining_hours} heure(s), {remaining_minutes} minute(s), et {remaining_seconds} seconde(s)")

# Fonctions de conversion
def convert_minutes(minutes):
    convert_seconds(minutes * 60)

def convert_hours(hours):
    convert_seconds(hours * 3600)

def convert_days(days):
    convert_seconds(days * 86400)

def convert_months(months):
    convert_seconds(months * 2592000)

def convert_years(years):
    convert_seconds(years * 31536000)

def convert_centuries(centuries):
    convert_seconds(centuries * 3155760000)

def convert_millennia(millennia):
    convert_seconds(millennia * 31557600000)

# Interface principale
def main():
    while True:
        print("\nOptions de conversion:")
        print("s - Secondes | m - Minutes | h - Heures")
        print("j - Jours | M - Mois | a - Années")
        print("c - Siècles | C - Millénaires | q - Quitter")
        
        choice = input("\nChoisissez l'unité (s/m/h/j/M/a/c/C/q): ").lower()
        
        if choice == 'q':
            print("Au revoir!")
            break
            
        if choice in ('s', 'm', 'h', 'j', 'm', 'a', 'c', 'c'):
            try:
                duration = float(input("Durée à convertir: "))
                
                if choice == 's': convert_seconds(duration)
                elif choice == 'm': convert_minutes(duration)
                elif choice == 'h': convert_hours(duration)
                elif choice == 'j': convert_days(duration)
                elif choice == 'm': convert_months(duration)
                elif choice == 'a': convert_years(duration)
                elif choice == 'c': convert_centuries(duration)
                elif choice == 'c': convert_millennia(duration)
                
            except ValueError:
                print("Veuillez entrer un nombre valide.")
        else:
            print("Option invalide. Veuillez réessayer.")

if __name__ == "__main__":
    main()

Ce script Python convertit une durée entre différentes unités temporelles (secondes jusqu'aux millénaires) et affiche le résultat décomposé.

Générateur de Mots de Passe Sécurisés

                                        import random
import string
import hashlib
import os

class PasswordGenerator:
    @staticmethod
    def generate_secure_password(length=12):
        """Génère un mot de passe complexe"""
        characters = string.ascii_letters + string.digits + string.punctuation
        while True:
            password = ''.join(random.choice(characters) for _ in range(length))
            if (any(c.isupper() for c in password) and
                any(c.islower() for c in password) and
                any(c.isdigit() for c in password) and
                any(c in string.punctuation for c in password)):
                return password

    @staticmethod
    def hash_password(password):
        """Applique SHA-256 au mot de passe"""
        return hashlib.sha256(password.encode()).hexdigest()

    @staticmethod
    def validate_password(password):
        """Valide les critères de sécurité"""
        if len(password) < 8:
            return "Le mot de passe doit contenir au moins 8 caractères."
        if not any(c.isupper() for c in password):
            return "Le mot de passe doit contenir au moins une majuscule."
        if not any(c.islower() for c in password):
            return "Le mot de passe doit contenir au moins une minuscule."
        if not any(c.isdigit() for c in password):
            return "Le mot de passe doit contenir au moins un chiffre."
        if not any(c in string.punctuation for c in password):
            return "Le mot de passe doit contenir au moins un caractère spécial."
        return None

class PasswordManager:
    def __init__(self):
        self.generator = PasswordGenerator()

    def run(self):
        while True:
            print("\n=== Gestionnaire de Mots de Passe ===")
            choice = input("1. Générer un mot de passe\n2. Valider un mot de passe\n3. Quitter\nChoix: ")
            
            if choice == '1':
                self.handle_generation()
            elif choice == '2':
                self.handle_validation()
            elif choice == '3':
                print("Merci d'avoir utilisé le programme. À bientôt !")
                break
            else:
                print("Option invalide. Veuillez réessayer.")
            
            if choice in ('1', '2'):
                input("\nAppuyez sur Entrée pour continuer...")
                self.clear_screen()

    def handle_generation(self):
        try:
            count = int(input("Nombre de suggestions à générer (3-20): "))
            count = max(3, min(20, count))
            length = int(input("Longueur du mot de passe (8-32): "))
            length = max(8, min(32, length))
            
            print(f"\nVoici {count} suggestions de mots de passe sécurisés:")
            suggestions = [self.generator.generate_secure_password(length) for _ in range(count)]
            
            for i, pwd in enumerate(suggestions, 1):
                print(f"{i}: {pwd}")
            
            self.process_password_choice(suggestions)
        except ValueError:
            print("Veuillez entrer un nombre valide.")

    def handle_validation(self):
        password = input("Entrez le mot de passe à valider: ")
        if error := self.generator.validate_password(password):
            print(f"\nErreur: {error}")
        else:
            print("\nMot de passe valide !")
            if input("Voulez-vous le hacher (SHA-256) ? (o/n): ").lower() == 'o':
                print(f"Hash: {self.generator.hash_password(password)}")

    def process_password_choice(self, suggestions):
        try:
            choice = int(input("\nChoisissez un mot de passe (numéro): "))
            if 1 <= choice <= len(suggestions):
                selected = suggestions[choice - 1]
                print(f"\nMot de passe sélectionné: {selected}")
                
                if input("Voulez-vous le hacher (SHA-256) ? (o/n): ").lower() == 'o':
                    print(f"Hash: {self.generator.hash_password(selected)}")
            else:
                print("Numéro invalide.")
        except ValueError:
            print("Veuillez entrer un numéro valide.")

    @staticmethod
    def clear_screen():
        os.system('cls' if os.name == 'nt' else 'clear')

if __name__ == "__main__":
    manager = PasswordManager()
    manager.run()

Automatisation Web avec Selenium

                                        from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.common.keys import Keys
from selenium.webdriver.support.ui import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
import keyboard
import time

class ArtStationAutomation:
    def __init__(self):
        self.driver = self.setup_driver()
        
    def setup_driver(self):
        """Configure les options du navigateur Chrome"""
        options = webdriver.ChromeOptions()
        options.add_argument("--disable-gpu")
        options.add_argument("--no-sandbox")
        options.add_argument("--disable-dev-shm-usage")
        options.add_argument("--start-maximized")
        options.add_experimental_option("excludeSwitches", ["enable-automation"])
        return webdriver.Chrome(options=options)

    def search_artstation(self):
        """Effectue une recherche sur ArtStation"""
        try:
            # Accéder à ArtStation
            self.driver.get("https://www.artstation.com/")
            
            # Recherche
            search_box = WebDriverWait(self.driver, 10).until(
                EC.presence_of_element_located((By.CSS_SELECTOR, "input[type='search']"))
            )
            search_box.send_keys("digital art")
            search_box.send_keys(Keys.RETURN)
            
            # Attendre les résultats
            WebDriverWait(self.driver, 15).until(
                EC.presence_of_element_located((By.CLASS_NAME, "project-card"))
            )
            print("Recherche effectuée avec succès")
            
        except Exception as e:
            print(f"Erreur lors de la recherche: {str(e)}")
            self.driver.save_screenshot("error_screenshot.png")

    def run(self):
        """Exécute l'automatisation"""
        print("Démarrage de l'automatisation...")
        self.search_artstation()
        
        print("\nAppuyez sur SUPPR pour quitter...")
        keyboard.wait("delete")
        
        self.driver.quit()
        print("Navigateur fermé.")

if __name__ == "__main__":
    automation = ArtStationAutomation()
    automation.run()
